  // system info (ikon seperti aslinya)
        const ua = navigator.userAgent
        document.getElementById('browser').textContent = ua.includes('Firefox')
          ? 'Firefox'
          : ua.includes('Chrome')
            ? 'Chrome'
            : ua.includes('Safari')
              ? 'Safari'
              : 'Browser'
        document.getElementById('os').textContent = /Windows/.test(ua)
          ? 'Windows'
          : /Mac/.test(ua)
            ? 'macOS'
            : /Linux/.test(ua)
              ? 'Linux'
              : /Android/.test(ua)
                ? 'Android'
                : /iPhone|iPad/.test(ua)
                  ? 'iOS'
                  : 'OS'

        fetch('https://ipapi.co/json/')
          .then((r) => r.json())
          .then((d) => {
            const flags = { ID: '🇮🇩', US: '🇺🇸', GB: '🇬🇧', JP: '🇯🇵', CN: '🇨🇳' }
            document.getElementById('loc').textContent =
              (flags[d.country_code] || '🌍') +
              ' ' +
              (d.city || d.region) +
              ', ' +
              d.country_name
          })
          .catch(() => {
            document.getElementById('loc').textContent = '🌍 Tidak diketahui'
          })
