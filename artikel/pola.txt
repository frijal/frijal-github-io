// --- DETEKSI INFO PENGUNJUNG ---
        const getBrowserInfo = () => {
            const ua = navigator.userAgent;
            if (ua.includes("Firefox")) return {name:"Firefox", icon:"https://upload.wikimedia.org/wikipedia/commons/a/a0/Firefox_logo%2C_2019.svg"};
            if (ua.includes("Edg")) return {name:"Edge", icon:"https://upload.wikimedia.org/wikipedia/commons/9/98/Microsoft_Edge_logo_%282019%29.svg"};
            if (ua.includes("Chrome")) return {name:"Chrome", icon:"https://upload.wikimedia.org/wikipedia/commons/a/a5/Google_Chrome_icon_%28September_2014%29.svg"};
            if (ua.includes("Safari")) return {name:"Safari", icon:"https://upload.wikimedia.org/wikipedia/commons/5/52/Safari_browser_logo.svg"};
            return {name:"Unknown", icon:""};
        };

        const getOSInfo = () => {
            const platform = navigator.platform.toLowerCase();
            if (platform.includes("win")) return {name:"Windows", icon:"https://upload.wikimedia.org/wikipedia/commons/e/e6/Windows_10_logo.svg"};
            if (platform.includes("mac")) return {name:"macOS", icon:"https://upload.wikimedia.org/wikipedia/commons/2/21/MacOS_logo_%282017%29.svg"};
            if (platform.includes("linux")) return {name:"Linux", icon:"https://upload.wikimedia.org/wikipedia/commons/3/35/Tux.svg"};
            if (/android/.test(navigator.userAgent.toLowerCase())) return {name:"Android", icon:"https://upload.wikimedia.org/wikipedia/commons/d/d7/Android_robot.svg"};
            if (/iphone|ipad|ipod/.test(navigator.userAgent.toLowerCase())) return {name:"iOS", icon:"https://upload.wikimedia.org/wikipedia/commons/f/fa/Apple_logo_black.svg"};
            return {name:"Unknown", icon:""};
        };

        const browser = getBrowserInfo();
        document.getElementById('browserIcon').src = browser.icon;
        document.getElementById('browserText').textContent = `Browser: ${browser.name}`;

        const os = getOSInfo();
        document.getElementById('osIcon').src = os.icon;
        document.getElementById('osText').textContent = `OS: ${os.name}`;

        fetch('https://ipapi.co/json/')
            .then(res => res.json())
            .then(data => {
                document.getElementById('ipText').textContent = `Lokasi: ${data.city}, ${data.country_name}`;
                document.getElementById('flagIcon').src = `https://flagcdn.com/w40/${data.country_code.toLowerCase()}.png`;
            })
            .catch(() => document.getElementById('ipText').textContent = 'Lokasi: Gagal dimuat');
