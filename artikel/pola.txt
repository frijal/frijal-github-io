// Deteksi Info Pengguna
        document.addEventListener('DOMContentLoaded', function() {
            // Browser & OS
            const ua = navigator.userAgent;
            const platform = navigator.platform;
            let browser = "Browser tidak dikenal";
            if (ua.includes("Firefox/")) browser = `Firefox üî•`;
            else if (ua.includes("Chrome/")) browser = `Chrome üöÄ`;
            else if (ua.includes("Safari/")) browser = `Safari üß≠`;
            document.getElementById('browser-info').textContent = `Browser: ${browser}`;
            
            let os = "OS tidak dikenal";
            if (platform.includes("Win")) os = "Windows üíª";
            else if (platform.includes("Mac")) os = "macOS üçé";
            else if (platform.includes("Linux")) os = "Linux üêß";
            document.getElementById('os-info').textContent = `Sistem Operasi: ${os}`;
            
            // Lokasi via IP
            const locationInfoEl = document.getElementById('location-info');
            fetch('https://ipapi.co/json/')
                .then(response => response.json())
                .then(data => {
                    const flag = `https://flagcdn.com/w40/${data.country_code.toLowerCase()}.png`;
                    locationInfoEl.innerHTML = `Lokasi: ${data.city}, ${data.country_name} <img src="${flag}" alt="${data.country_name} flag" style="width: 20px; vertical-align: middle;">`;
                })
                .catch(error => {
                    locationInfoEl.textContent = 'üìç Lokasi tidak dapat dideteksi.';
